CREATE PROCEDURE `deletePokemon`(IN idPokemon INT)
   BEGIN
		
        DELETE FROM userpokemon where numPokemon = idPokemon;
        
        DELETE FROM pokemon.pokemons WHERE numPokemon = idPokemon;
        
    END $$
